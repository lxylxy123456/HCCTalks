\begin{frame} [fragile]
	\frametitle{字符串方法}
	\linespread{1.5}
	\begin{lstlisting}[style=pythonstyle, gobble=4, texcl, escapechar=@]
	"HCC, I'm. ".upper()	# 大写
	"HCC, I'm. ".lower()	# 小写
	"HCC, I'm. ".split()	# 按空白字符分割
	len("HCC, I'm. ")		# 得到长度
	'&'.join([1, 2])		# 合并
	@\color[HTML]{ff0bff}\textquotesingle\textquotesingle@'Hello from HCC:
	Welcome!
	'''.split('\n')			# 按换行符分割
	\end{lstlisting}
\end{frame}
